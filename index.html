Volvé a ingresar el PIN.', code: 'SESSION_EXPIRED' }); return responder(eliminar(data)); case 'getCouriers': return responder(getCouriersList()); case 'saveCouriers': return responder(saveCouriersList(data.couriers)); const sheet = obtenerHoja(ss); const rows = sheet.getDataRange().getValues(); const hoy = fechaHoy(); const sobres = rows.slice(1) .filter(r => { if (!r[COLS.ID - 1] || r[COLS.ID - 1] === '') return false; if (couriers.length > 0) { hoja.getRange(2, 1, couriers.length, 1).setValues(couriers.map(c => [c])); } Logger.log('guardado OK'); return { ok: true }; } function registrarLog(nombre, accion, courier, notas) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); hoja.getRange(1,1,1,16).setFontWeight('bold').setBackground('#1A1916').setFontColor('#FFFFFF'); } return hoja; } function fechaHoy() { return Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'dd/MM/yyyy');
